	add_to_variable = {
		var = global.london_treaty_credibility
		value = -4
	}
	clamp_variable = {
		var = global.london_treaty_credibility
		min = 0
		max = 5
	}

treaty_credibility_change_3 = {
	IF = {
		limit = { 
			OR = {
				tag = FRA
				tag = ITA
			} 
		}
		add_to_variable = {
			var = global.london_treaty_credibility
			value = -2
		}
	}
	ELSE = {
		add_to_variable = {
			var = global.london_treaty_credibility
			value = -1
		}
	}
	clamp_variable = {
		var = global.london_treaty_credibility
		min = 0
		max = 5
	}
}
